		\frametitle{Knots and links in the 3-sphere}
		\begin{definition}
			A \textit{knot} $K$ is the image of a smooth embedding of the circle $S^1$ into the $3$-sphere $S^3$. In particular, $K$ is diffeomorphic to $S^1$. A \textit{link} $L$ is a disjoint union of knots.
		\end{definition}
		\begin{definition}
		An \textit{ambient isotopy} of $S^3$ between embeddings $g\!:S^1\to S^3$ and $h\!:S^1\to S^3$ is a continuous map $F\!:S^3\times[0,1]\to S^3$, such that $F_t=F(\cdot,t)$ is a homeomorphism for each $t\in[0,1]$,  $F_0=\mathbbm{1}$, and $F_1\circ g = h$.
		\end{definition}
		\begin{itemize}
		\item We regard two knots $K,K'\subset S^3$ to be equivalent if they differ by an ambient isotopy of $S^3$. We write $K\simeq K'$.
		\end{itemize}
